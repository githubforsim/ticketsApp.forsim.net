9/************************************************
10 * Contrôleur LoginController gérant les actions liées à l'authentification des utilisateurs.
11 *
12 * Ce contrôleur offre des méthodes pour gérer l'authentification des utilisateurs,
13 * y compris la connexion, la déconnexion et la gestion des demandes de changement de mot de passe.
14 * Il utilise le modèle UserModel pour interagir avec la base de données et exécuter les opérations
15 * nécessaires à l'authentification.
16 *
17 * Les fonctionnalités spécifiques incluent :
18 * - Connexion d'un utilisateur avec validation des identifiants et gestion de session.
19 * - Déconnexion d'un utilisateur en détruisant la session active.
20 * - Gestion des demandes de changement de mot de passe en vérifiant l'existence de l'utilisateur.
21 *
22 * Ce contrôleur exploite les sessions PHP pour maintenir l'état de connexion des utilisateurs,
23 * redirigeant vers différentes vues en fonction du rôle de l'utilisateur (administrateur ou utilisateur standard).
24 * Les redirections sont gérées via des en-têtes HTTP pour diriger l'utilisateur vers les pages appropriées
25 * après une action d'authentification.
26 ***********************************************/

62 /************************************************
63 * Fonction de connexion utilisateur.
64 *
65 * @param string $username Le nom d'utilisateur à valider et à nettoyer
66 * @param string $password Le mot de passe à valider et à nettoyer
67 ***********************************************/
68 public function login($username, $password)
69 {
70 //Valide et nettoie les données
71 $username = trim($username);
72 $password = trim($password);
73
74 //Vérifie si une session est cours, si non alors création d'une
75 if (session_status() == PHP_SESSION_NONE) {
76 session_start();
77 }
78 $user = $this->UserModel->getUserByUsername($username);
79
80 //Si les identifiants sont corrects
81 if ($user && password_verify($password, $user['pwd'])) {
82 $_SESSION['username'] = $username;
83 $_SESSION['loggedIn'] = true;
84
85 $role = $this->UserModel->getUserRole($username);
86 // Définir le rôle de l'utilisateur dans la session
87 $_SESSION['role'] = $role;
88
89 // Redirirection vers la page d'administration si l'utilisateur est un administrateur
90 if ($role === 'admin') {
91 header('Location: /ticketsApp/app/src/Views/admin/admin_view.php');
92 exit();
93 } else {
94 header('Location: /ticketsApp/app/src/Views/index.php');
95 exit();
96 }
97
98 } else {
99 header('Location: /ticketsApp/app/src/Views/login_view.php?erreur=1');
100 exit();
101 }

136 // Rediriger vers une page de confirmation si la demande de changement de mot de passe est valide
